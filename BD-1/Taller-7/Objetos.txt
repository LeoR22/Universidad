CREATE DATABASE OBJETOSVIRTUALES;
use objetosvirtuales;

create table facultades(
nombre varchar(30) not null,
codigo int not null primary key) ;

create table departamentos(
nombre varchar(30) not null,
codigo int not null primary key,
cod_facultad int not null,
constraint fk_dep foreign key(cod_facultad)
references facultades(codigo)); 

create table temas (
codigo int not null primary key,
nombre varchar(40) not null); 

create table profesores (
nombre varchar(40) not null,
email varchar(30) null,
cedula int not null primary key,
cod_dpto int not null,
constraint fk_prof foreign key (cod_dpto) 
references departamentos(codigo));
